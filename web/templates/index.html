    // ── Feature 1: Hello World ──
    async function sayHello() {
      const name = document.getElementById("name-input").value.trim();
      if (!name) { alert("Please enter your name first!"); return; }
      document.getElementById("hello-loading").style.display = "block";
      document.getElementById("hello-result").style.display = "none";
      try {
        const res = await fetch("/hello", {
          method: "POST",
          headers: { "Content-Type": "application/json" },
          body: JSON.stringify({ name: name })
        });
        const data = await res.json();
        const resultBox = document.getElementById("hello-result");
        resultBox.textContent = data.message;
        resultBox.style.display = "block";
      } catch (error) {
        document.getElementById("hello-result").textContent = "⚠️ Error: Could not reach the agent. Is the server running?";
        document.getElementById("hello-result").style.display = "block";
      } finally {
        document.getElementById("hello-loading").style.display = "none";
      }
    }

    // ── Feature 2: Generate Prompt ──
    async function generatePrompt() {
      const task = document.getElementById("task-input").value.trim();
      if (!task) { alert("Please describe a task first!"); return; }
      document.getElementById("prompt-loading").style.display = "block";
      document.getElementById("prompt-result").style.display = "none";
      try {
        const res = await fetch("/generate-prompt", {
          method: "POST",
          headers: { "Content-Type": "application/json" },
          body: JSON.stringify({ task })
        });
        const data = await res.json();
        document.getElementById("prompt-output").textContent = data.prompt || data.error;
        document.getElementById("prompt-filename").textContent = data.filename ? "💾 Saved as: " + data.filename : "";
        document.getElementById("prompt-result").style.display = "block";
      } catch (error) {
        document.getElementById("prompt-output").textContent = "⚠️ Error: Could not reach the agent. Is the server running?";
        document.getElementById("prompt-result").style.display = "block";
      } finally {
        document.getElementById("prompt-loading").style.display = "none";
      }
    }

    function copyPrompt() {
      const text = document.getElementById("prompt-output").textContent;
      navigator.clipboard.writeText(text).then(() => {
        const btn = document.querySelector(".copy-btn");
        btn.textContent = "Copied!";
        setTimeout(() => btn.textContent = "Copy", 2000);
      });
    }

    function showCategory() {
      var cat = document.getElementById("bp-category").value;
      var container = document.getElementById("bp-list");
      if (!cat) { container.style.display = "none"; return; }
      var items = BEST_PRACTICES[cat] || [];
      container.innerHTML = items
        .map(function(p, i) {
          return "

";
        })
        .join("");
      container.style.display = "block";
    }

    // ── Feature 4: Badminton Coach ──
    const BC_MODES = ["daily-drill","tip-of-day","match-analysis","footwork-coach","weekly-plan"];

    function bcOnModeChange() {
      const mode = document.getElementById("bc-mode").value;
      BC_MODES.forEach(m => {
        const el = document.getElementById("bc-inputs-" + m);
        if (el) el.style.display = (m === mode) ? (m === "daily-drill" || m === "footwork-coach" || m === "match-analysis" || m === "weekly-plan" ? "grid" : "block") : "none";
      });
      document.getElementById("bc-result").style.display = "none";
    }

    function renderCoachOutput(text) {
      return text.split('\n').map(line => {
        const t = line.trim();
        if (!t) return '
';
        if (/^[A-Z][A-Z\s—&:()\d]+$/.test(t) && t.length > 3)
          return `
${t}
`;
        if (t.startsWith('•'))
          return `
${t}
`;
        if (/^(Step|Drill)\s\d/.test(t))
          return `
${t}
`;
        return `
${t}
`;
      }).join('');
    }

    async function badmintonCoach() {
      const mode = document.getElementById("bc-mode").value;
      let payload = { mode };

      if (mode === "daily-drill") {
        payload.focus   = document.getElementById("bc-drill-focus").value;
        payload.level   = document.getElementById("bc-drill-level").value;
        payload.minutes = document.getElementById("bc-drill-time").value;
      } else if (mode === "tip-of-day") {
        payload.focus = document.getElementById("bc-tip-focus").value;
      } else if (mode === "match-analysis") {
        payload.worked   = document.getElementById("bc-match-worked").value.trim();
        payload.didnt    = document.getElementById("bc-match-didnt").value.trim();
        payload.opponent = document.getElementById("bc-match-opponent").value.trim();
        if (!payload.worked && !payload.didnt) { alert("Please describe what worked or what didn't in the match."); return; }
      } else if (mode === "footwork-coach") {
        payload.pattern = document.getElementById("bc-fw-pattern").value;
        payload.level   = document.getElementById("bc-fw-level").value;
      } else if (mode === "weekly-plan") {
        payload.sessions   = document.getElementById("bc-plan-sessions").value;
        payload.goal       = document.getElementById("bc-plan-goal").value;
        payload.weaknesses = document.getElementById("bc-plan-weaknesses").value.trim();
      }

      const btn = document.getElementById("bc-generate-btn");
      btn.disabled = true; btn.textContent = "Generating…";
      document.getElementById("bc-loading").style.display = "block";
      document.getElementById("bc-result").style.display = "none";

      try {
        const res = await fetch("/badminton-coach", {
          method: "POST",
          headers: { "Content-Type": "application/json" },
          body: JSON.stringify(payload)
        });
        const data = await res.json();
        if (data.error) { alert(data.error); return; }
        document.getElementById("bc-output").innerHTML = renderCoachOutput(data.result);
        document.getElementById("bc-result").style.display = "block";
      } catch (err) {
        alert("Error connecting to Badminton Coach.");
      } finally {
        document.getElementById("bc-loading").style.display = "none";
        btn.disabled = false; btn.textContent = "🏸 Generate";
      }
    }
